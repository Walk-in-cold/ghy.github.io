  1. function [Lmn Tmn]= from_eigenvector_out_am(v)  

  2. Ndim=length(v);  

  3. Lmn=[];  

  4. Tmn=zeros(Ndim);  

  5. for m=1:Ndim-1  

  6. for n=m+1:Ndim   

  7. vm=v(m);  

  8. vn=v(n);  

  9. area_m_n = abs(vm)*abs(vn)*pi*sin(angle(vm)-angle(vn));  

  10. Lmn=[Lmn; area_m_n];  

  11. Tmn(m,n) = area_m_n;  

  12. Tmn(n,m) = -area_m_n;  

  13. end;  

  14. end  

  15. end
